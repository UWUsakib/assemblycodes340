TASK-1


.MODEL SMALL
 
.STACK 100H

.DATA

; declare variables here 
 first db 'ALPHABET: $'
 SECOND DB 'NUMBER: $'
 THIRD DB 'RESULT: $'

.CODE
MAIN PROC

; initialize DS

MOV AX,@DATA
MOV DS,AX
 
; enter your code here 

    LEA DX,first
    MOV AH,9
    INT 21H
    
    MOV AH,1
    INT 21H
    MOV BL,AL
    
    MOV AH,2
    MOV DL,10
    INT 21H
    MOV AH,2
    MOV DL,13
    INT 21H 
    
    
    LEA DX,SECOND
    MOV AH,9
    INT 21H
    
    MOV AH,1
    INT 21H
    MOV CL,AL
    SUB CL,48
    
    
    
    MOV AH,2
    MOV DL,10
    INT 21H
    MOV AH,2
    MOV DL,13
    INT 21H    
    
    
    LEA DX,THIRD
    MOV AH,9
    INT 21H
    
    ADD BL,CL
    
    
    MOV AH,2
    MOV DL,BL  
    INT 21H
    




 

;exit to DOS
               
MOV AX,4C00H
INT 21H

MAIN ENDP
    END MAIN
